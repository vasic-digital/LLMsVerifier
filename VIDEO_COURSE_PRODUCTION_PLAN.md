# Screen recording configuration
SCREEN_WIDTH=1920
SCREEN_HEIGHT=1080
FPS=30

AUDIO_CODEC="aac"

# Create recording directories
mkdir -p ~/recordings/llm-verifier-course

  -vcodec $VIDEO_CODEC -preset ultrafast \

  ~/recordings/llm-verifier-course/raw/lecture1.mkv

### 3. Progress Tracking
```javascript
// llm-verifier/video-platform/interactive/progress.js
class VideoProgressTracker {
    constructor(videoId) {
        this.videoId = videoId;
        this.watchedSegments = new Set();
        this.loadProgress();
    }
    
    trackProgress(currentTime, duration) {
        const segment = Math.floor(currentTime / 10); // 10-second segments
        this.watchedSegments.add(segment);
        this.saveProgress();
    }
    
    getCompletionPercentage() {
        const totalSegments = Math.ceil(this.duration / 10);
        return (this.watchedSegments.size / totalSegments) * 100;
    }
    
    saveProgress() {
        localStorage.setItem(`progress-${this.videoId}`, 
            JSON.stringify(Array.from(this.watchedSegments)));
    }
    
    loadProgress() {
        const saved = localStorage.getItem(`progress-${this.videoId}`);
        if (saved) {
            this.watchedSegments = new Set(JSON.parse(saved));
        }
    }
}
```
