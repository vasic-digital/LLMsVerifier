Getting Started

# Clone the repository
git clone https://github.com/vasic-digital/LLMsVerifier.git
cd LLMsVerifier/llm-verifier

# Build the application
go build -o llm-verifier cmd/main.go

# Run verification
./llm-verifier verify --all

# Export verified configurations
./llm-verifier export --format opencode --output ~/Downloads/

Docker Installation

# Start with Docker Compose
docker-compose up -d

# Access the web interface at http://localhost:8080
